 32      >>> nx.is_eulerian(nx.complete_graph(5)) 
 33      True 
 34      >>> nx.is_eulerian(nx.petersen_graph()) 
 35      False 
 36   
 37      Notes 
 38      ----- 
 39      This implementation requires the graph to be connected 
 40      (or strongly connected for directed graphs). 
 41      """ 
 42      if G.is_directed(): 
 43          # Every node must have equal in degree and out degree 
 44          for n in G.nodes_iter(): 
 45              if G.in_degree(n) != G.out_degree(n): 
 46                 return False 
 47          # Must be strongly connected 
 48          if not nx.is_strongly_connected(G): 
 49              return False 
 50      else: 
 51          # An undirected Eulerian graph has no vertices of odd degrees 
 52          for v,d in G.degree_iter(): 
 53              if d % 2 != 0: 
 54                  return False 
 55          # Must be connected 
 56          if not nx.is_connected(G): 
 57              return False 
 58      return True
