 24      Parameters

 31         A dictionary keyed by node with a 1 or 0 as data for each node color. 
 32   
 33      Raises 
 34      ------ 
 35      NetworkXError if the graph is not two-colorable. 
 36   
 37      Examples 
 38      -------- 
 39      >>> from networkx.algorithms import bipartite 
 40      >>> G = nx.path_graph(4) 
 41      >>> c = bipartite.color(G) 
 42      >>> print(c) 
 43      {0: 1, 1: 0, 2: 1, 3: 0} 
 44   
 45      You can use this to set a node attribute indicating the biparite set: 
 46       
 47      >>> nx.set_node_attributes(G, 'bipartite', c) 
 48      >>> print(G.node[0]['bipartite']) 
 49      1 
 50      >>> print(G.node[1]['bipartite']) 
 51      0 
 52      """ 
 53      color = {} 
 54      for n in G: # handle disconnected graphs 
 55          if n in color or len(G[n])==0: # skip isolates 
 56              continue 
 57          queue = [n]   
 58          color[n] = 1 # nodes seen with color (1 or 0) 
 59          while queue: 
 60              v = queue.pop() 
 61              c = 1 - color[v] # opposite color of node v 
 62              for w in G[v]:  
 63                  if w in color:  
 64                      if color[w] == color[v]: 
 65                          raise nx.NetworkXError("Graph is not bipartite.") 
 66                  else: 
 67                      color[w] = c 
 68                      queue.append(w) 
 69      # color isolates with 0 
 70      color.update(dict.fromkeys(nx.isolates(G),0))
