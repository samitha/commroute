 14 -def blockmodel(G,partitions,multigraph=False): 

 15      """Returns a reduced graph constructed using the generalized block modeling 
 16      technique. 
 17   
 18      The blockmodel technique collapses nodes into blocks based on a 
 19      given partitioning of the node set.  Each partition of nodes 
 20      (block) is represented as a single node in the reduced graph. 
 21   
 22      Edges between nodes in the block graph are added according to the 
 23      edges in the original graph.  If the parameter multigraph is False 
 24      (the default) a single edge is added with a weight equal to the 
 25      sum of the edge weights between nodes in the original graph 
 26      The default is a weight of 1 if weights are not specified.  If the 
 27      parameter multigraph is True then multiple edges are added each 
 28      with the edge data from the original graph. 
 29   
 30      Parameters 
 31      ---------- 
 32      G : graph 
 33          A networkx Graph or DiGraph 
 34      partitions : list of lists, or list of sets  
 35          The partition of the nodes.  Must be non-overlapping. 
 36      multigraph : bool, optional 
 37          If True return a MultiGraph with the edge data of the original 
 38          graph applied to each corresponding edge in the new graph. 
 39          If False return a Graph with the sum of the edge weights, or a 
 40          count of the edges if the original graph is unweighted. 
 41   
 42      Returns 
 43      ------- 
 44      blockmodel : a Networkx graph object 
 45       
 46      Examples 
 47      -------- 
 48      >>> G=nx.path_graph(6) 
 49      >>> partition=[[0,1],[2,3],[4,5]] 
 50      >>> M=nx.blockmodel(G,partition) 
 51   
 52      References 
 53      ---------- 
 54      .. [1] Patrick Doreian, Vladimir Batagelj, and Anuska Ferligoj 
 55          "Generalized Blockmodeling",Cambridge University Press, 2004. 
 56      """ 
 57      # Create sets of node partitions 
 58      part=list(map(set,partitions))  
 59   
 60      # Check for overlapping node partitions 
 61      u=set() 
 62      for p1,p2 in zip(part[:-1],part[1:]): 
 63          u.update(p1) 
 64          #if not u.isdisjoint(p2):  # Python 2.6 required 
 65          if len (u.intersection(p2))>0: 
 66              raise nx.NetworkXException("Overlapping node partitions.") 
 67   
 68      # Initialize blockmodel graph 
 69      if multigraph: 
 70          if G.is_directed(): 
 71              M=nx.MultiDiGraph()  
 72          else: 
 73              M=nx.MultiGraph()  
 74      else: 
 75          if G.is_directed(): 
 76              M=nx.DiGraph()  
 77          else: 
 78              M=nx.Graph()  
 79           
 80      # Add nodes and properties to blockmodel             
 81      # The blockmodel nodes are node-induced subgraphs of G 
 82      # Label them with integers starting at 0 
 83      for i,p in zip(range(len(part)),part): 
 84          M.add_node(i) 
 85          # The node-induced subgraph is stored as the node 'graph' attribute  
 86          SG=G.subgraph(p) 
 87          M.node[i]['graph']=SG         
 88          M.node[i]['nnodes']=SG.number_of_nodes() 
 89          M.node[i]['nedges']=SG.number_of_edges() 
 90          M.node[i]['density']=nx.density(SG) 
 91           
 92      # Create mapping between original node labels and new blockmodel node labels 
 93      block_mapping={} 
 94      for n in M: 
 95          nodes_in_block=M.node[n]['graph'].nodes() 
 96          block_mapping.update(dict.fromkeys(nodes_in_block,n)) 
 97   
 98      # Add edges to block graph  
 99      for u,v,d in G.edges(data=True): 
100          bmu=block_mapping[u] 
101          bmv=block_mapping[v] 
102          if bmu==bmv: # no self loops 
103              continue 
104          if multigraph:  
105              # For multigraphs add an edge for each edge in original graph 
106              M.add_edge(bmu,bmv,attr_dict=d) 
107          else: 
108              # For graphs and digraphs add single weighted edge 
109              weight=d.get('weight',1.0) # default to 1 if no weight specified 
110              if M.has_edge(bmu,bmv): 
111                  M[bmu][bmv]['weight']+=weight 
112              else: 
113                  M.add_edge(bmu,bmv,weight=weight) 
114      return M
