 56      """ 
 57      Return the small graph described by graph_description. 
 58   
 59      graph_description is a list of the form [ltype,name,n,xlist] 
 60   
 61      Here ltype is one of "adjacencylist" or "edgelist", 
 62      name is the name of the graph and n the number of nodes. 
 63      This constructs a graph of n nodes with integer labels 0,..,n-1. 
 64       
 65      If ltype="adjacencylist"  then xlist is an adjacency list 
 66      with exactly n entries, in with the j'th entry (which can be empty) 
 67      specifies the nodes connected to vertex j. 
 68      e.g. the "square" graph C_4 can be obtained by 
 69   
 70      >>> G=nx.make_small_graph(["adjacencylist","C_4",4,[[2,4],[1,3],[2,4],[1,3]]]) 
 71   
 72      or, since we do not need to add edges twice, 
 73       
 74      >>> G=nx.make_small_graph(["adjacencylist","C_4",4,[[2,4],[3],[4],[]]]) 
 75       
 76      If ltype="edgelist" then xlist is an edge list  
 77      written as [[v1,w2],[v2,w2],...,[vk,wk]], 
 78      where vj and wj integers in the range 1,..,n 
 79      e.g. the "square" graph C_4 can be obtained by 
 80    
 81      >>> G=nx.make_small_graph(["edgelist","C_4",4,[[1,2],[3,4],[2,3],[4,1]]]) 
 82   
 83      Use the create_using argument to choose the graph class/type.  
 84      """ 
 85      ltype=graph_description[0] 
 86      name=graph_description[1] 
 87      n=graph_description[2] 
 88   
 89      G=empty_graph(n, create_using) 
 90      nodes=G.nodes() 
 91   
 92      if ltype=="adjacencylist": 
 93          adjlist=graph_description[3] 
 94          if len(adjlist) != n: 
 95              raise NetworkXError("invalid graph_description") 
 96          G.add_edges_from([(u-1,v) for v in nodes for u in adjlist[v]]) 
 97      elif ltype=="edgelist": 
 98          edgelist=graph_description[3] 
 99          for e in edgelist: 
100              v1=e[0]-1 
101              v2=e[1]-1 
102              if v1<0 or v1>n-1 or v2<0 or v2>n-1: 
103                  raise NetworkXError("invalid graph_description") 
104              else: 
105                  G.add_edge(v1,v2) 
106      G.name=name 
107      return G

111      """ 
112      Return the cubic graph specified in LCF notation. 
113   
114      LCF notation (LCF=Lederberg-Coxeter-Fruchte) is a compressed 
115      notation used in the generation of various cubic Hamiltonian 
116      graphs of high symmetry. See, for example, dodecahedral_graph, 
117      desargues_graph, heawood_graph and pappus_graph below. 
118       
119      n (number of nodes) 
120        The starting graph is the n-cycle with nodes 0,...,n-1. 
121        (The null graph is returned if n < 0.) 
122   
123      shift_list = [s1,s2,..,sk], a list of integer shifts mod n, 
124   
125      repeats 
126        integer specifying the number of times that shifts in shift_list 
127        are successively applied to each v_current in the n-cycle 
128        to generate an edge between v_current and v_current+shift mod n. 
129   
130      For v1 cycling through the n-cycle a total of k*repeats 
131      with shift cycling through shiftlist repeats times connect 
132      v1 with v1+shift mod n 
133             
134      The utility graph K_{3,3} 
135   
136      >>> G=nx.LCF_graph(6,[3,-3],3) 
137       
138      The Heawood graph 
139   
140      >>> G=nx.LCF_graph(14,[5,-5],7) 
141   
142      See http://mathworld.wolfram.com/LCFNotation.html for a description 
143      and references. 
144       
145      """ 
146      if create_using is not None and create_using.is_directed(): 
147          raise NetworkXError("Directed Graph not supported") 
148   
149      if n <= 0: 
150          return empty_graph(0, create_using) 
151   
152      # start with the n-cycle 
153      G=cycle_graph(n, create_using) 
154      G.name="LCF_graph" 
155      nodes=G.nodes() 
156   
157      n_extra_edges=repeats*len(shift_list)     
158      # edges are added n_extra_edges times 
159      # (not all of these need be new) 
160      if n_extra_edges < 1: 
161          return G 
162   
163      for i in range(n_extra_edges): 
164          shift=shift_list[i%len(shift_list)] #cycle through shift_list 
165          v1=nodes[i%n]                    # cycle repeatedly through nodes 
166          v2=nodes[(i + shift)%n] 
167          G.add_edge(v1, v2) 
168      return G
