 21      The eccentricity of a node v is the maximum distance from v to

 24      Parameters 
 25      ---------- 
 26      G : NetworkX graph

 28   
 29      v : node, optional 
 30         Return value of specified node        
 31   
 32      sp : dict of dicts, optional        
 33         All pairs shortest path lenghts as a dictionary of dictionaries 
 34   
 35      Returns 
 36      ------- 
 37      ecc : dictionary 
 38         A dictionary of eccentricity values keyed by node. 
 39      """ 
 40      nodes=[] 
 41      if v is None:                # none, use entire graph  
 42          nodes=G.nodes()  
 43      elif isinstance(v, list):  # check for a list 
 44          nodes=v 
 45      else:                      # assume it is a single value 
 46          nodes=[v] 
 47      order=G.order() 
 48   
 49      e={} 
 50      for v in nodes: 
 51          if sp is None: 
 52              length=networkx.single_source_shortest_path_length(G,v)

 55          try: 
 56              L = len(length)

 58              raise networkx.NetworkXError('Format of "sp" is invalid.') 
 59          else: 
 60              if L != order:
