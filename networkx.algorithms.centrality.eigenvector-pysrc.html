147      if type(G) == nx.MultiGraph or type(G) == nx.MultiDiGraph:

155      A=nx.adj_matrix(G,nodelist=G.nodes()) 
156      eigenvalues,eigenvectors=np.linalg.eig(A) 
157      # eigenvalue indices in reverse sorted order 
158      ind=eigenvalues.argsort()[::-1]
