 88      algorithm. 
 89       
 90      This algorithm uses Edmonds-Karp-Dinitz path selection rule which 
 91      guarantees a running time of O(nm^2) for n nodes and m edges. 
 92   
 93   
 94      Parameters 
 95      ---------- 
 96      G : NetworkX graph 
 97          Edges of the graph are expected to have an attribute called 
 98          'capacity'. If this attribute is not present, the edge is 
 99          considered to have infinite capacity. 
100   
101      s : node 
102          Source node for the flow. 
103   
104      t : node 
105          Sink node for the flow. 
106   
107      capacity: string 
108          Edges of the graph G are expected to have an attribute capacity 
109          that indicates how much flow the edge can support. If this 
110          attribute is not present, the edge is considered to have 
111          infinite capacity. Default value: 'capacity'. 
112   
113      Returns 
114      ------- 
115      flow_value : integer, float 
116          Value of the maximum flow, i.e., net outflow from the source. 
117   
118      flow_dict : dictionary 
119          Dictionary of dictionaries keyed by nodes such that 
120          flow_dict[u][v] is the flow edge (u, v). 
121   
122      Raises 
123      ------ 
124      NetworkXError 
125          The algorithm does not support MultiGraph and MultiDiGraph. If 
126          the input graph is an instance of one of these two classes, a 
127          NetworkXError is raised. 
128   
129      NetworkXUnbounded 
130          If the graph has a path of infinite capacity, the value of a  
131          feasible flow on the graph is unbounded above and the function 
132          raises a NetworkXUnbounded. 
133   
134      Examples 
135      -------- 
136      >>> import networkx as nx 
137      >>> G = nx.DiGraph() 
138      >>> G.add_edge('x','a', capacity=3.0) 
139      >>> G.add_edge('x','b', capacity=1.0) 
140      >>> G.add_edge('a','c', capacity=3.0) 
141      >>> G.add_edge('b','c', capacity=5.0) 
142      >>> G.add_edge('b','d', capacity=4.0) 
143      >>> G.add_edge('d','e', capacity=2.0) 
144      >>> G.add_edge('c','y', capacity=2.0) 
145      >>> G.add_edge('e','y', capacity=3.0) 
146      >>> flow, F = nx.ford_fulkerson(G, 'x', 'y') 
147      >>> flow 
148      3.0 
149      """ 
150      if G.is_multigraph(): 
151          raise nx.NetworkXError( 
152                  'MultiGraph and MultiDiGraph not supported (yet).') 
153   
154      auxiliary, inf_capacity_flows = _create_auxiliary_digraph(G, 
155                                                          capacity=capacity) 
156      flow_value = 0   # Initial feasible flow. 
157   
158      # As long as there is an (s, t)-path in the auxiliary digraph, find 
159      # the shortest (with respect to the number of arcs) such path and 
160      # augment the flow on this path. 
161      while True: 
162          try: 
163              path_nodes = nx.bidirectional_shortest_path(auxiliary, s, t) 
164          except nx.NetworkXNoPath: 
165              break 
166   
167          # Get the list of edges in the shortest path. 
168          path_edges = list(zip(path_nodes[:-1], path_nodes[1:])) 
169   
170          # Find the minimum capacity of an edge in the path. 
171          try: 
172              path_capacity = min([auxiliary[u][v][capacity] 
173                                  for u, v in path_edges 
174                                  if capacity in auxiliary[u][v]]) 
175          except ValueError:  
176              # path of infinite capacity implies no max flow 
177              raise nx.NetworkXUnbounded( 
178                      "Infinite capacity path, flow unbounded above.") 
179           
180          flow_value += path_capacity 
181   
182          # Augment the flow along the path. 
183          for u, v in path_edges: 
184              edge_attr = auxiliary[u][v] 
185              if capacity in edge_attr: 
186                  edge_attr[capacity] -= path_capacity 
187                  if edge_attr[capacity] == 0: 
188                      auxiliary.remove_edge(u, v) 
189              else: 
190                  inf_capacity_flows[(u, v)] += path_capacity 
191   
192              if auxiliary.has_edge(v, u): 
193                  if capacity in auxiliary[v][u]: 
194                      auxiliary[v][u][capacity] += path_capacity 
195              else: 
196                  auxiliary.add_edge(v, u, {capacity: path_capacity}) 
197       
198      flow_dict = _create_flow_dict(G, auxiliary, inf_capacity_flows, 
199                                    capacity=capacity) 
200      return flow_value, flow_dict
