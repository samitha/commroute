 478  d3.format = function(specifier) { 
 479    var match = d3_format_re.exec(specifier), 
 480        fill = match[1] || " ", 
 481        sign = match[3] || "", 
 482        zfill = match[5], 
 483        width = +match[6], 
 484        comma = match[7], 
 485        precision = match[8], 
 486        type = match[9], 
 487        percentage = false, 
 488        integer = false; 
 489   
 490    if (precision) precision = precision.substring(1); 
 491   
 492    if (zfill) { 
 493      fill = "0"; // TODO align = "="; 
 494      if (comma) width -= Math.floor((width - 1) / 4); 
 495    } 
 496   
 497    switch (type) { 
 498      case "n": comma = true; type = "g"; break; 
 499      case "%": percentage = true; type = "f"; break; 
 500      case "p": percentage = true; type = "r"; break; 
 501      case "d": integer = true; precision = "0"; break; 
 502    } 
 503   
 504    type = d3_format_types[type] || d3_format_typeDefault; 
 505   
 506    return function(value) { 
 507      var number = percentage ? value * 100 : +value, 
 508          negative = (number < 0) && (number = -number) ? "\u2212" : sign; 
 509   
 510      // Return the empty string for floats formatted as ints. 
 511      if (integer && (number % 1)) return ""; 
 512   
 513      // Convert the input value to the desired precision. 
 514      value = type(number, precision); 
 515   
 516      // If the fill character is 0, the sign and group is applied after the fill. 
 517      if (zfill) { 
 518        var length = value.length + negative.length; 
 519        if (length < width) value = new Array(width - length + 1).join(fill) + value; 
 520        if (comma) value = d3_format_group(value); 
 521        value = negative + value; 
 522      } 
 523   
 524      // Otherwise (e.g., space-filling), the sign and group is applied before. 
 525      else { 
 526        if (comma) value = d3_format_group(value); 
 527        value = negative + value; 
 528        var length = value.length; 
 529        if (length < width) value = new Array(width - length + 1).join(fill) + value; 
 530      } 
 531      if (percentage) value += "%"; 
 532   
 533      return value; 
 534    };

1193    var m1, 
1194        m2;

1202    m2 = l <= .5 ? l * (1 + s) : l + s - l * s;

1209      if (h < 180) return m2; 
1210      if (h < 240) return m1 + (m2 - m1) * (240 - h) / 60;
